JavaScript Objects

Objects are collections of key-value pairs. They allow you to group related data and functionality together.

Creating Objects

Use curly braces {} to create an object:

// Empty object
let emptyObj = {};

// Object with properties
let person = {
    name: "John",
    age: 25,
    city: "St. Louis"
};

// Objects can hold any type of value
let student = {
    firstName: "Alice",
    lastName: "Smith",
    age: 20,
    gpa: 3.8,
    isEnrolled: true,
    courses: ["HTML", "CSS", "JavaScript"]
};

console.log(student);

Accessing Object Properties

Dot Notation

let person = {
    name: "John",
    age: 25,
    city: "St. Louis"
};

console.log(person.name);  // "John"
console.log(person.age);   // 25
console.log(person.city);  // "St. Louis"

Bracket Notation

let person = {
    name: "John",
    age: 25,
    "favorite color": "blue"  // Property with spaces
};

console.log(person["name"]);  // "John"
console.log(person["age"]);   // 25

// Necessary for properties with spaces
console.log(person["favorite color"]); // "blue"

// Useful with variables
let prop = "name";
console.log(person[prop]); // "John"

Modifying Objects

let car = {
    brand: "Toyota",
    model: "Camry",
    year: 2020
};

// Change existing property
car.year = 2021;
console.log(car.year); // 2021

// Add new property
car.color = "blue";
console.log(car.color); // "blue"

// Delete property
delete car.model;
console.log(car); // { brand: "Toyota", year: 2021, color: "blue" }

Object Methods

Objects can contain functions, called methods:

let person = {
    firstName: "John",
    lastName: "Doe",
    age: 25,
    
    // Method using regular function
    greet: function() {
        console.log("Hello, my name is " + this.firstName);
    },
    
    // Shorthand method syntax (ES6)
    getFullName() {
        return this.firstName + " " + this.lastName;
    },
    
    haveBirthday() {
        this.age++;
        console.log("Happy birthday! Now " + this.age);
    }
};

person.greet();           // "Hello, my name is John"
console.log(person.getFullName()); // "John Doe"
person.haveBirthday();    // "Happy birthday! Now 26"

Note: The this keyword refers to the object that the method belongs to. It allows methods to access other properties and methods of the same object.

Nested Objects

Objects can contain other objects:

let student = {
    name: "Alice",
    age: 20,
    address: {
        street: "123 Main St",
        city: "St. Louis",
        state: "MO",
        zip: "63103"
    },
    grades: {
        math: 95,
        english: 88,
        science: 92
    }
};

// Access nested properties
console.log(student.address.city);      // "St. Louis"
console.log(student.grades.math);       // 95
console.log(student.address.street);    // "123 Main St"

Looping Through Objects

For...in Loop

let person = {
    name: "John",
    age: 25,
    city: "St. Louis"
};

// Loop through all properties
for (let key in person) {
    console.log(key + ": " + person[key]);
}
// Output:
// name: John
// age: 25
// city: St. Louis

Object.keys(), Object.values(), Object.entries()

let student = {
    name: "Alice",
    age: 20,
    gpa: 3.8
};

// Get all keys
let keys = Object.keys(student);
console.log(keys); // ["name", "age", "gpa"]

// Get all values
let values = Object.values(student);
console.log(values); // ["Alice", 20, 3.8]

// Get key-value pairs
let entries = Object.entries(student);
console.log(entries);
// [["name", "Alice"], ["age", 20], ["gpa", 3.8]]

// Loop through entries
Object.entries(student).forEach(([key, value]) => {
    console.log(`${key}: ${value}`);
});

Object Destructuring

Extract properties into variables:

let person = {
    name: "John",
    age: 25,
    city: "St. Louis"
};

// Extract properties
let { name, age, city } = person;

console.log(name);  // "John"
console.log(age);   // 25
console.log(city);  // "St. Louis"

// With different variable names
let { name: personName, age: personAge } = person;
console.log(personName); // "John"
console.log(personAge);  // 25
