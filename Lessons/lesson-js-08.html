JavaScript Arrays

Arrays are used to store multiple values in a single variable. They're one of the most useful data structures in JavaScript.

Creating Arrays

// Array of strings
let fruits = ["apple", "banana", "orange"];

// Array of numbers
let numbers = [1, 2, 3, 4, 5];

// Mixed types (not recommended, but possible)
let mixed = ["text", 42, true, null];

// Empty array
let empty = [];

console.log(fruits); // ["apple", "banana", "orange"]

Accessing Array Elements

Arrays use zero-based indexing (first element is at index 0):

let colors = ["red", "green", "blue"];

console.log(colors[0]); // "red" (first element)
console.log(colors[1]); // "green" (second element)
console.log(colors[2]); // "blue" (third element)

// Get array length
console.log(colors.length); // 3

// Access last element
console.log(colors[colors.length - 1]); // "blue"

Modifying Arrays

let fruits = ["apple", "banana", "orange"];

// Change an element
fruits[1] = "grape";
console.log(fruits); // ["apple", "grape", "orange"]

// Add element at specific index
fruits[3] = "mango";
console.log(fruits); // ["apple", "grape", "orange", "mango"]

Array Methods

Push and Pop

Add or remove elements from the end:

let stack = [1, 2, 3];

// push() - add to end
stack.push(4);
console.log(stack); // [1, 2, 3, 4]

stack.push(5, 6);
console.log(stack); // [1, 2, 3, 4, 5, 6]

// pop() - remove from end
let last = stack.pop();
console.log(last);  // 6
console.log(stack); // [1, 2, 3, 4, 5]

Shift and Unshift

Add or remove elements from the beginning:

let queue = [2, 3, 4];

// unshift() - add to beginning
queue.unshift(1);
console.log(queue); // [1, 2, 3, 4]

// shift() - remove from beginning
let first = queue.shift();
console.log(first); // 1
console.log(queue); // [2, 3, 4]

Slice

Extract a portion of an array (doesn't modify original):

let numbers = [1, 2, 3, 4, 5];

// slice(start, end) - end is exclusive
let slice1 = numbers.slice(1, 4);
console.log(slice1);   // [2, 3, 4]
console.log(numbers);  // [1, 2, 3, 4, 5] (unchanged)

// slice(start) - from start to end
let slice2 = numbers.slice(2);
console.log(slice2); // [3, 4, 5]

// slice() - copy entire array
let copy = numbers.slice();
console.log(copy); // [1, 2, 3, 4, 5]

Splice

Add or remove elements at any position (modifies original):

let letters = ["a", "b", "c", "d", "e"];

// splice(index, deleteCount, items...)
// Remove 2 elements starting at index 1
letters.splice(1, 2);
console.log(letters); // ["a", "d", "e"]

// Add elements at index 1
letters.splice(1, 0, "b", "c");
console.log(letters); // ["a", "b", "c", "d", "e"]

// Replace elements
letters.splice(2, 1, "x");
console.log(letters); // ["a", "b", "x", "d", "e"]

IndexOf and Includes

Search for elements in an array:

let fruits = ["apple", "banana", "orange", "banana"];

// indexOf() - returns first index, or -1 if not found
console.log(fruits.indexOf("banana")); // 1
console.log(fruits.indexOf("grape"));  // -1

// lastIndexOf() - returns last index
console.log(fruits.lastIndexOf("banana")); // 3

// includes() - returns true/false
console.log(fruits.includes("apple"));  // true
console.log(fruits.includes("grape"));  // false

Looping Through Arrays

For Loop

let numbers = [10, 20, 30, 40];

for (let i = 0; i < numbers.length; i++) {
    console.log(numbers[i]);
}

For...of Loop

let colors = ["red", "green", "blue"];

for (let color of colors) {
    console.log(color);
}
// Output: red, green, blue

ForEach Method

let fruits = ["apple", "banana", "orange"];

fruits.forEach(function(fruit, index) {
    console.log(index + ": " + fruit);
});
// Output:
// 0: apple
// 1: banana
// 2: orange

// Arrow function version
fruits.forEach((fruit) => {
    console.log(fruit);
});
